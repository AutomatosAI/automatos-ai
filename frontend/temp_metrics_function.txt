  async getSystemMetrics(): Promise<Record<string, any>> {
    try {
      // Try to get real metrics from API
      return await this.request('/api/system/metrics');
    } catch (error) {
      // API not available - return basic fallback data
      console.warn('System metrics API not available, using fallback data');
      return {
        cpu: {
          average_usage: 25.0
        },
        memory: {
          percent: 45.0
        },
        network: {
          packets_sent: 1000,
          packets_recv: 1200
        }
      };
    }
  }
